# ТЕХНИЧЕСКОЕ ЗАДАНИЕ
## Калькулятор диетотерапии при нарушениях митохондриального β-окисления жирных кислот (FAOD)

**Версия:** 1.0
**Дата:** 2024
**Основа:** Клинические рекомендации РФ 2024, МКБ E71.3

---

## 1. ОБЩИЕ СВЕДЕНИЯ

### 1.1. Назначение системы
Веб-калькулятор для расчёта индивидуальной диетотерапии пациентов с наследственными нарушениями митохондриального β-окисления жирных кислот (FAOD).

### 1.2. Целевая аудитория
- Врачи-генетики
- Врачи-диетологи
- Врачи-педиатры
- Врачи-неонатологи

### 1.3. Ключевые требования
1. Расчёт диеты для всех 11 форм FAOD
2. Учёт возраста, массы тела, клинической формы, текущего состояния
3. Генерация документа для печати/пересылки (Word/PDF)
4. Интуитивно понятный интерфейс
5. Работа без подключения к интернету (после загрузки)

- Единственная форма или форма не влияет на расчёт диеты

### 5.1. Расчёт возраста

```javascript
function calculateAge(birthDate, currentDate = new Date()) {
  const years = currentDate.getFullYear() - birthDate.getFullYear();
  const months = currentDate.getMonth() - birthDate.getMonth();
  const days = currentDate.getDate() - birthDate.getDate();

  let ageYears = years;
  let ageMonths = months;

  if (days < 0) ageMonths--;
  if (ageMonths < 0) {
    ageYears--;
    ageMonths += 12;
  }

  return { years: ageYears, months: ageMonths, totalMonths: ageYears * 12 + ageMonths };
}
```

### 5.2. Расчёт суточной потребности в энергии

```javascript
function calculateDailyEnergy(weightKg, ageMonths, sex, currentState) {
  // 1. Получить норму ккал/кг из справочника по возрасту и полу
  const energyPerKg = getEnergyNorm(ageMonths, sex);

  // 2. Базовая потребность
  let dailyEnergy = weightKg * energyPerKg;

  // 3. Коррекция при кризе (+10-20%)
  if (currentState === 'CRISIS' || currentState === 'CRISIS_SUSPECTED') {
    dailyEnergy *= 1.15; // +15%
  } else if (currentState === 'INTERCURRENT') {
    dailyEnergy *= 1.10; // +10%
  }

  return Math.round(dailyEnergy);
}
```

### 5.3. Расчёт макронутриентов

```javascript
function calculateMacros(dailyEnergy, diagnosisId, clinicalForm, currentState) {
  // Получить соотношение из справочника
  const ratios = getMacroRatios(diagnosisId, clinicalForm);

  // Расчёт в граммах
  const result = {
    // Жиры: 1 г = 9 ккал
    fatTotal: {
      percent: ratios.fatPercent,
      grams: Math.round((dailyEnergy * ratios.fatPercent / 100) / 9),
      kcal: Math.round(dailyEnergy * ratios.fatPercent / 100)
    },

    // MCT: 1 г = 8.3 ккал
    mct: {
      percent: ratios.mctPercent,
      grams: Math.round((dailyEnergy * ratios.mctPercent / 100) / 8.3),
      kcal: Math.round(dailyEnergy * ratios.mctPercent / 100)
    },

    // LCT: 1 г = 9 ккал
    lct: {
      percent: ratios.lctPercent,
      grams: Math.round((dailyEnergy * ratios.lctPercent / 100) / 9),
      kcal: Math.round(dailyEnergy * ratios.lctPercent / 100)
    },

    // Углеводы: 1 г = 4 ккал
    carbs: {
      percent: ratios.carbsPercent,
      grams: Math.round((dailyEnergy * ratios.carbsPercent / 100) / 4),
      kcal: Math.round(dailyEnergy * ratios.carbsPercent / 100)
    },

    // Белки: 1 г = 4 ккал
    protein: {
      percent: ratios.proteinPercent,
      grams: Math.round((dailyEnergy * ratios.proteinPercent / 100) / 4),
      kcal: Math.round(dailyEnergy * ratios.proteinPercent / 100)
    }
  };

  return result;
}
```

```javascript
function calculateFeedingSchedule(ageMonths, currentState) {
  // Базовые интервалы
  let schedule = getFeedingIntervals(ageMonths);

  // Коррекция при болезни
  if (currentState === 'INTERCURRENT') {
    schedule.dayInterval = Math.max(2, schedule.dayInterval - 1);
    schedule.nightInterval = Math.max(3, schedule.nightInterval - 2);
  } else if (currentState === 'CRISIS' || currentState === 'CRISIS_SUSPECTED') {
    schedule.dayInterval = 2;
    schedule.nightInterval = 2;
  }

  // Количество кормлений
  const awakeHours = 14; // примерно
  const sleepHours = 10;

  schedule.feedingsPerDay = Math.ceil(awakeHours / schedule.dayInterval) +
                            Math.ceil(sleepHours / schedule.nightInterval);

  return schedule;
}
```

    glucoseRate = 7.7;
  } else if (ageMonths < 12) {
    glucoseConcentration = 15;
    glucoseRate = 7.0;
  } else if (ageMonths < 36) {
    glucoseConcentration = 20;
    glucoseRate = 4.5;
  } else {
    glucoseConcentration = 25;
    glucoseRate = 3.3;
  }

  protocol.homeActions = [
    `Дать ${glucoseConcentration}% раствор глюкозы/декстрозы`,
    `Доза: ${glucoseRate} мл/кг/час = ${Math.round(glucoseRate * weightKg)} мл/час`,
    'Исключить жиры полностью на 24-48 часов',
    'Кормить каждые 2-3 часа'
  ];

  protocol.hospitalizationCriteria = [
    'Многократная рвота',
    'Отказ от питья',
    'Нарушение сознания',
    'Судороги',
    'Гипогликемия < 2.5 ммоль/л'
  ];

  // В/в терапия
  let ivGlucoseRate;
  if (ageMonths < 36) ivGlucoseRate = '10-12 мг/кг/мин';
  else if (ageMonths < 120) ivGlucoseRate = '8-10 мг/кг/мин';
  else ivGlucoseRate = '5-8 мг/кг/мин';

  protocol.ivTherapy = {
    glucose: `10% декстроза в/в ${ivGlucoseRate}`,
    bicarbonate: 'При pH < 7.2 или HCO3 < 16 — натрия гидрокарбонат',
    ammonia: 'При NH3 > 150-200 мкмоль/л — натрия бензоат 250-500 мг/кг/сут'
  };

  return protocol;
}
